// MediFast Secure Firebase Rules v1.0
// Last updated: 2026 — DO NOT MODIFY without understanding!

rules_version = '2';
service cloud.firestore {
  match /databases/{database}/documents {

    // ── HELPER FUNCTIONS ──
    function isLoggedIn() {
      return request.auth != null;
    }

    function isOwner(userId) {
      return request.auth.uid == userId;
    }

    function isAdmin() {
      return isLoggedIn() &&
        request.auth.token.admin == true;
    }

    // ── USERS COLLECTION ──
    // Patients can ONLY read/write their own data
    // No one else can see patient data!
    match /users/{userId} {
      allow read: if isLoggedIn() && isOwner(userId);
      allow create: if isLoggedIn() && isOwner(userId);
      allow update: if isLoggedIn() && isOwner(userId);
      allow delete: if isAdmin();
    }

    // ── PHARMACIES COLLECTION ──
    // Anyone logged in can read pharmacy basic info
    // Only the pharmacy owner can edit their data
    match /pharmacies/{pharmacyId} {
      allow read: if isLoggedIn();
      allow create: if isLoggedIn() && isOwner(pharmacyId);
      allow update: if isLoggedIn() &&
        (isOwner(pharmacyId) || isAdmin());
      allow delete: if isAdmin();

      // Medicines — patients can search, pharmacy manages
      match /medicines/{medicineId} {
        allow read: if isLoggedIn();
        allow write: if isLoggedIn() && isOwner(pharmacyId);
      }

      // Orders — only patient who placed & pharmacy see it
      match /orders/{orderId} {
        allow read: if isLoggedIn() &&
          (isOwner(pharmacyId) ||
           request.auth.uid == resource.data.patientId);
        allow create: if isLoggedIn();
        allow update: if isLoggedIn() && isOwner(pharmacyId);
        allow delete: if isAdmin();
      }

      // Prescriptions — MOST SENSITIVE — strict access!
      match /prescriptions/{rxId} {
        allow read: if isLoggedIn() &&
          (isOwner(pharmacyId) ||
           request.auth.uid == resource.data.patientId);
        allow create: if isLoggedIn() &&
          request.auth.uid == request.resource.data.patientId;
        allow update: if isLoggedIn() && isOwner(pharmacyId);
        allow delete: if isAdmin();
      }

      // Purchases — pharmacy only
      match /purchases/{purchaseId} {
        allow read, write: if isLoggedIn() && isOwner(pharmacyId);
      }
    }

    // ── COMPLAINTS ──
    // Anyone can submit, only admin can delete
    match /complaints/{complaintId} {
      allow create: if isLoggedIn();
      allow read: if isLoggedIn() &&
        (request.auth.uid == resource.data.userId || isAdmin());
      allow update: if isAdmin();
      allow delete: if isAdmin();
    }

    // ── MASTER MEDICINES — public read, admin write ──
    match /masterMedicines/{medicineId} {
      allow read: if isLoggedIn();
      allow write: if isAdmin();
    }

    // ── ANNOUNCEMENTS — public read, admin write ──
    match /announcements/{announcementId} {
      allow read: if isLoggedIn();
      allow write: if isAdmin();
    }

    // ── APP SETTINGS — admin only ──
    match /appSettings/{setting} {
      allow read: if isLoggedIn();
      allow write: if isAdmin();
    }

    // ── AUDIT LOGS — write only, admin read ──
    // No one can ever delete audit logs!
    match /auditLogs/{logId} {
      allow create: if isLoggedIn();
      allow read: if isAdmin();
      allow update, delete: if false; // NEVER deleteable!
    }

    // ── BLOCK EVERYTHING ELSE ──
    match /{document=**} {
      allow read, write: if false;
    }
  }
}

Part 2 — Storage Security Rules
Protects prescription images and documents
⚠️ Important
1
Go to Firebase → Storage → Rules
Replace existing rules with these secure rules:
rules_version = '2';
service firebase.storage {
  match /b/{bucket}/o {

    // Prescription images — VERY SENSITIVE
    match /prescriptions/{userId}/{filename} {
      // Only the patient who uploaded can read
      allow read: if request.auth != null &&
        request.auth.uid == userId;
      // Only authenticated users can upload
      allow write: if request.auth != null &&
        request.auth.uid == userId &&
        request.resource.size < 5 * 1024 * 1024 && // Max 5MB
        request.resource.contentType.matches('image/.*|application/pdf');
    }

    // Pharmacy documents
    match /pharmacy-docs/{pharmacyId}/{filename} {
      allow read: if request.auth != null;
      allow write: if request.auth != null &&
        request.auth.uid == pharmacyId &&
        request.resource.size < 10 * 1024 * 1024; // Max 10MB
    }

    // Block everything else
    match /{allPaths=**} {
      allow read, write: if false;
    }
  }
}

// 1. MASK SENSITIVE DATA
function maskPhone(phone) {
  if (!phone) return '-';
  return phone.replace(
    /(\+91|0)?(\d{3})(\d{4})(\d{3})/,
    '$1$2****$4'
  );
}

function maskEmail(email) {
  if (!email) return '-';
  const [user, domain] = email.split('@');
  return user.slice(0,2) + '****@' + domain;
}

function maskAadhar(num) {
  if (!num) return '-';
  return 'XXXX-XXXX-' + num.slice(-4);
}

// 2. AUDIT LOGGER — logs every action
async function auditLog(db, action, details) {
  try {
    await addDoc(collection(db, 'auditLogs'), {
      action,
      details,
      userId: auth.currentUser?.uid,
      userEmail: auth.currentUser?.email,
      timestamp: serverTimestamp(),
      ip: 'client-side',
      userAgent: navigator.userAgent.slice(0, 100)
    });
  } catch(e) {
    console.error('Audit log failed:', e);
  }
}

// 3. RATE LIMITER — prevents spam
const rateLimiter = {
  attempts: {},
  check(key, maxAttempts = 5, windowMs = 60000) {
    const now = Date.now();
    if (!this.attempts[key]) this.attempts[key] = [];
    this.attempts[key] = this.attempts[key].filter(
      t => now - t < windowMs
    );
    if (this.attempts[key].length >= maxAttempts) {
      return false; // Rate limited!
    }
    this.attempts[key].push(now);
    return true;
  }
};

// 4. INPUT SANITIZER — prevents XSS attacks
function sanitize(input) {
  if (!input) return '';
  return String(input)
    .replace(/&/g, '&')
    .replace(/'<')
    .replace(/>/g, '>')
    .replace(/"/g, '"')
    .replace(/'/g, ''')
    .trim();
}

// 5. SESSION CHECKER — auto logout after inactivity
let inactivityTimer;
function resetInactivityTimer() {
  clearTimeout(inactivityTimer);
  inactivityTimer = setTimeout(() => {
    // Auto logout after 30 minutes of inactivity
    if (auth.currentUser) {
      alert('Session expired for security. Please login again.');
      signOut(auth).then(() => {
        window.location.href = 'medifast-auth.html';
      });
    }
  }, 30 * 60 * 1000); // 30 minutes
}

// Start inactivity tracking
['click', 'keypress', 'scroll', 'mousemove'].forEach(event => {
  document.addEventListener(event, resetInactivityTimer);
});
resetInactivityTimer();

// 6. HTTPS ENFORCER
if (location.protocol !== 'https:' &&
    location.hostname !== 'localhost') {
  location.replace(
    'https:' + window.location.href.substring(
      window.location.protocol.length
    )
  );
}

// 8. PRESCRIPTION SECURITY
// Generates secure temporary URL for prescription
function getSecurePrescriptionUrl(url, expiryMinutes = 60) {
  // In production: generate signed URL with expiry
  // For now: return URL with timestamp validation
  const expiry = Date.now() + (expiryMinutes * 60 * 1000);
  localStorage.setItem('rx_expiry_' + btoa(url), expiry);
  return url;
}

function validatePrescriptionAccess(url) {
  const expiry = localStorage.getItem('rx_expiry_' + btoa(url));
  if (!expiry || Date.now() > parseInt(expiry)) {
    return false; // Expired!
  }
  return true;
}

// 9. DATA ENCRYPTION HELPERS
function encryptSensitive(data) {
  // Base64 encode for basic obfuscation
  // In production: use proper AES encryption
  return btoa(encodeURIComponent(JSON.stringify(data)));
}

function decryptSensitive(encoded) {
  try {
    return JSON.parse(decodeURIComponent(atob(encoded)));
  } catch { return null; }
}

// 10. SECURITY HEADERS CHECK
function checkSecurityHeaders() {
  const checks = {
    https: location.protocol === 'https:',
    localStorage: typeof localStorage !== 'undefined',
    cookies: navigator.cookieEnabled
  };
  console.log('🛡️ Security checks:', checks);
  return checks;
}
